# T0G Tournament Bot - Server Command Cheat Sheet

# Start bot
systemctl start tog_tournament_bot

# Stop bot
systemctl stop tog_tournament_bot

# Restart bot (use this after updating code)
systemctl restart tog_tournament_bot

# Check bot status
systemctl status tog_tournament_bot   # press Q to exit

# View live logs (real-time)
journalctl -u tog_tournament_bot -f   # press CTRL+C to stop watching

# View full log history
journalctl -u tog_tournament_bot

# Reload systemd (only if you edit the service file)
systemctl daemon-reload
